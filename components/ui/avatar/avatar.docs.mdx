import { Avatar } from './avatar'

A component that renders an avatar with animated borders

### Component usage

```tsx live
<div style={{ padding: 50, backgroundColor: 'black' }}>
  <Avatar
    alt={'avatar'}
    src={'https://avatars.githubusercontent.com/u/6298488?v=4'}
    width={200}
    colors={[
      '#fb0094',
      '#0000ff',
      '#00ff00',
      '#ffff00',
      '#ff0000',
      '#fb0094',
      '#0000ff',
      '#00ff00',
      '#ffff00',
      '#ff0000'
    ]}
    animationDuration={20}
    borderBlur={20}
    borderSize={4}
  />
</div>
```
